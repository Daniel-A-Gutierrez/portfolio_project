<CodeSnippet filename="feedback.js">
```js
    #!/usr/bin/env node

    /**
    * Frontend Feedback Loop Tool
    * 
    * Captures screenshots and DOM metrics for frontend iteration feedback.
    * Generates a feedback report in the feedback/ directory.
    */

    import { mkdirSync, writeFileSync } from 'fs';
    import { join, dirname } from 'path';
    import { fileURLToPath } from 'url';
    import http from 'http';
    import { exec } from 'child_process';
    import playwright from 'playwright';

    const __filename = fileURLToPath(import.meta.url);
    const __dirname = dirname(__filename);
    const projectRoot = join(__dirname, '..');
    const feedbackDir = join(projectRoot, 'feedback');

    // Configuration
    const CONFIG = {
    devServerPort: 4321,
    devServerUrl: 'http://localhost:4321',
    timeout: 30000,
    screenshotFormat: 'png'
    };

    /**
    * Check if dev server is running
    */
    function isServerRunning() {
    return new Promise((resolve) => {
        const url = new URL(CONFIG.devServerUrl);
        const protocol = url.protocol === 'https:' ? https : http;
        
        const req = protocol.request({
        hostname: 'localhost',
        port: CONFIG.devServerPort,
        timeout: 2000,
        method: 'GET',
        path: '/'
        }, (res) => {
        res.resume();
        resolve(true);
        });

        req.on('error', () => resolve(false));
        req.on('timeout', () => {
        req.destroy();
        resolve(false);
        });
        req.end();
    });
    }

    /**
    * Start the Astro dev server
    */
    function startDevServer() {
    return new Promise((resolve, reject) => {
        console.log('Starting Astro dev server...');
        
        // Check if already running first
        isServerRunning().then((running) => {
        if (running) {
            console.log('Dev server is already running on port', CONFIG.devServerPort);
            resolve(null);
            return;
        }

        const child = exec('yarn dev', {
            cwd: projectRoot,
            detached: true,
            stdio: 'ignore'
        });

        child.unref();

        // Wait for server to be ready
        let attempts = 0;
        const maxAttempts = 15;

        const checkServer = () => {
            attempts++;
            console.log(`Waiting for dev server... (${attempts}/${maxAttempts})`);
            
            isServerRunning().then((running) => {
            if (running) {
                console.log('Dev server is ready!');
                resolve(child);
            } else if (attempts >= maxAttempts) {
                reject(new Error('Dev server failed to start'));
            } else {
                setTimeout(checkServer, 1000);
            }
            });
        };

        setTimeout(checkServer, 2000);
        });
    });
    }

    /**
    * Stop the dev server
    */
    function stopDevServer(server) {
    if (server && server.pid) {
        console.log('Stopping dev server...');
        try {
        process.kill(-server.pid);
        } catch (e) {
        // Ignore errors
        }
    }
    }

    /**
    * Extract DOM metrics from the page
    */
    function extractDomMetrics(page) {
        ...// lots of console.logging , abridging for blog brevity
    }

    /**
    * Capture screenshots
    */
    async function captureScreenshots(page, screenshotName) {
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const filename = `${screenshotName}-${timestamp}`;
    
    // Full page screenshot
    const fullPath = join(feedbackDir, `${filename}-full.${CONFIG.screenshotFormat}`);
    await page.screenshot({
        path: fullPath,
        fullPage: true
    });
    console.log(`Full page screenshot saved: ${fullPath}`);

    // Viewport screenshot
    const viewportPath = join(feedbackDir, `${filename}-viewport.${CONFIG.screenshotFormat}`);
    await page.screenshot({
        path: viewportPath,
        fullPage: false
    });
    console.log(`Viewport screenshot saved: ${viewportPath}`);

    return {
        fullPage: fullPath,
        viewport: viewportPath
    };
    }

    /**
    * Generate feedback report
    */
    function generateReport(metrics, screenshots, timestamp) {
    const report = {
        timestamp,
        url: CONFIG.devServerUrl,
        metrics,
        screenshots,
        summary: {
        elementCount: metrics.totalElements,
        imageCount: metrics.images.count,
        linkCount: metrics.links,
        buttonCount: metrics.buttons,
        maxDomDepth: metrics.maxDomDepth,
        totalTextLength: metrics.totalTextLength,
        viewportDimensions: `${metrics.viewport.width}x${metrics.viewport.height}`,
        documentDimensions: `${metrics.documentSize.width}x${metrics.documentSize.height}`
        }
    };

    const reportPath = join(feedbackDir, `feedback-report-${timestamp}.json`);
    writeFileSync(reportPath, JSON.stringify(report, null, 2));
    console.log(`\nFeedback report saved: ${reportPath}`);

    return report;
    }

    /**
    * Print report summary to console
    */
    function printSummary(report) { 
        //more logging, abridging for blod.
    }

    /**
    * Main function
    */
    async function main() {
    const timestamp = new Date().toISOString();
    let serverProcess = null;

    // Ensure feedback directory exists
    mkdirSync(feedbackDir, { recursive: true });

    console.log('Frontend Feedback Loop Tool');
    console.log('===========================\n');

    try {
        // Start dev server if not running
        serverProcess = await startDevServer();

        // Give server a moment to fully initialize
        await new Promise(resolve => setTimeout(resolve, 2000));

        // Launch browser
        console.log('Launching browser...');
        const browser = await playwright.chromium.launch({
        headless: true
        });

        const context = await browser.newContext({
        viewport: { width: 1920, height: 1080 }
        });

        const page = await context.newPage();

        // Navigate to homepage
        console.log(`Navigating to ${CONFIG.devServerUrl}...`);
        await page.goto(CONFIG.devServerUrl, {
        waitUntil: 'networkidle',
        timeout: CONFIG.timeout
        });

        // Wait for any animations to complete
        await page.waitForTimeout(1000);

        // Capture screenshots
        console.log('\nCapturing screenshots...');
        const screenshots = await captureScreenshots(page, 'homepage');

        // Extract DOM metrics
        console.log('Extracting DOM metrics...');
        const metrics = await extractDomMetrics(page);

        // Generate report
        const report = generateReport(metrics, screenshots, timestamp);

        // Print summary to console
        printSummary(report);

        // Close browser
        await browser.close();

        console.log('\nFeedback collection complete!');
        console.log(`View screenshots and full report in: ${feedbackDir}`);

    } catch (error) {
        console.error('Error during feedback collection:', error.message);
        if (error.stack) {
        console.error(error.stack);
        }
        process.exit(1);
    } finally {
        // Stop dev server if we started it
        stopDevServer(serverProcess);
    }
    }

    main();

```
