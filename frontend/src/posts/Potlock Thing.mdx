        #[rustfmt::ignore]
        pub fn db_append(conn: &Connection, key: &str, val: &str) -> Result<(), DBError>
        {
            if val.len() > 10_000 { return Err(DBError::Oversized); }
            let mut b4 = db_get(conn,key)?.ok_or_else(|| DBError::KeyDoesntExist)?;
            b4.value += val;
            if b4.key.len() + b4.value.len() > 20_000 {return Err(DBError::Oversized);}
            conn.execute("UPDATE storage SET value = $1 WHERE key = $2;",(b4.value,b4.key)) //doesnt support limit!
                .map_err(|e| DBError::DBError(e))?;
            return Ok(());
        }

        pub fn db_delete(conn: &Connection, key: &str) -> Result<(), DBError>
        {
            if key.len() > 10_000 {return Err(DBError::Oversized);}
            conn.query_one("SELECT * FROM storage WHERE key = $1 LIMIT 1;", (key,), |row| {
                            Ok(KVRow::try_from(row).expect("Row did not match table schema"))
                        })
                        .optional()
                        .map_err(|e| DBError::DBError(e))?
                        .ok_or_else(|| DBError::KeyDoesntExist)?;
            conn.execute("DELETE FROM storage WHERE key = $1;",(key,))
                .map_err(|e| DBError::DBError(e))?;
            return Ok(());
        }

# Debugging

Some silliness trying to talk to the wrong server then discovering a wrong base url for kv_get, which should have been 
/api/kv_get not api/kv_get. 

KV_SET returned a 422 status code - a first for me. inspecting the network tab it seems the issue is that value is expected
to be a string, not json. 
The raw request from the browser is : 
<CodeSnippet filename = "the request">
    ``` json 
        {"key":"ab65c54a-6835-4313-ad86-55e4103739dc","value":{"uuid":"ab65c54a-6835-4313-ad86-55e4103739dc","name":"Fake Party","date":"2002-12-12","time":"12:12","location":"here"}}
    ```
</CodeSnippet>

Easy fix, in KVSet \{key,value\} needs to be \{key, value : JSON.stringify(value)\}; 

Also, 
- fetch json needed a sibling function that doesnt expect to return anything, 
- Commit needed to actually be bound to the button, 
- Everything needed to use svelte's new $state(value) rune, 
- And some other little mistakes.

But yeah! Potlucks!
