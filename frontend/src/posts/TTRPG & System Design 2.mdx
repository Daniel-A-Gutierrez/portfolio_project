    #[derive(PartialEq,Clone)]
    enum DamageType {
        True,
        Fire,
        All
    }

    #[derive(Clone)]
    struct Damage { 
        amounts : Vec<(DamageType, i64)>
    }

    //maybe I don't need a trait since everything gets damaged the same way?
    // pub trait Damageable { 
    //     //emit an event
    //     fn damage(&mut self, damage : Damage);
    // }
    enum DefenseType {
        Flat(i64),
        Ratio(f64) // multiply incoming damage by ratio
    }

    pub struct Defenses {
        defenses : Vec<(DamageType,DefenseType)>,
        hp : i64,
        max_hp : i64,
        guard : i64,
        barrier : i64
    }

    impl Defenses {
        fn damage(&mut self, incoming : &Damage) {
            let mut dmg = incoming.clone();
            for (dmg_type1,deftype) in &self.defenses {
                for (dmg_type2,amt) in &mut dmg.amounts {
                    if dmg_type1 != dmg_type2 { continue ; }
                    *amt = match deftype {
                        DefenseType::Flat(f) => (*amt - f).clamp(0, i64::MAX),
                        DefenseType::Ratio(r) => (*amt as f64 * r).clamp(0.0, i64::MAX as f64) as i64
                    }
                }
            }
        }
    }

    fn damage(&mut self, incoming : &Damage) {
        let mut dmg = incoming.clone();
        for (dmg_type1,deftype) in &self.defenses {
            for (dmg_type2,amt) in &mut dmg.amounts {
                if *dmg_type1 != DamageType::All && dmg_type1 != dmg_type2 { continue ; }
                *amt = match deftype {
                    DefenseType::Flat(f) => (*amt - f).clamp(0, i64::MAX),
                    DefenseType::Ratio(r) => (*amt as f64 * r).clamp(0.0, i64::MAX as f64) as i64
                }
            }
        }
        self.hp -= dmg.amounts.iter().fold(0, |a,b| a+b.1);
    }
