        True,

    enum DefenseType {
        Flat(i64),
        Ratio(f64) // multiply incoming damage by ratio
    }

    pub struct Defenses {
        defenses : Vec<(DamageType,DefenseType)>,
        hp : i64,
        max_hp : i64,
        guard : i64,
        barrier : i64
    }

    impl Defenses {
        fn damage(&mut self, incoming : &Damage) {
            let mut dmg = incoming.clone();
            for (dmg_type1,deftype) in &self.defenses {
                for (dmg_type2,amt) in &mut dmg.amounts {
                    if dmg_type1 != dmg_type2 { continue ; }
                    *amt = match deftype {
                        DefenseType::Flat(f) => (*amt - f).clamp(0, i64::MAX),
                        DefenseType::Ratio(r) => (*amt as f64 * r).clamp(0.0, i64::MAX as f64) as i64
                    }
                }
            }
        }
    }

            }

        Attack(Damage),

    #[derive(Debug)]
    pub struct Event {
        source : u64,
        targets : Vec<u64>,
        data : EventData
    }

    pub struct EventListener{
        order : i64,
        pub matcher : Box<dyn Fn(&Event) -> bool>,
        pub operator : Box<dyn Fn(&mut World, Event) -> Option<Event>> 
    }

    pub struct EventSystem 
    {
        turn_order : VecDeque<u64>,
        world : World, 
        listeners : Vec<EventListener>,
        turn_counter : u64
        //plans : Vec
    }

    //an event system could be a thing thats constructed on combat start
    //from a subset of units in the world.
    impl EventSystem {
        fn start_combat(&mut self) {
            self.turn_counter = 0;
            //everything in the world will now engage in a battle to the death
            self.turn_order = self.world.actors.keys().map(|k|*k).collect::<VecDeque<u64>>();
            //lets just run till theres only one living unit
            while self.turn_order.len() > 1{
                let turn = self.turn_order.pop_front().unwrap();
                let mut actor = self.world.actors.get(&turn).expect("Actor with turn does not exist");
                if(0 > actor.defenses.hp) { continue ; }
                let events = actor.act(&self.world);
                for event in events {
                    dbg!(&event);
                    self.dispatch(event);
                }
                //need to relookup to shadow previous actor to make borrow checker happy.
                let mut actor = self.world.actors.get(&turn).expect("Actor with turn does not exist");
                if( actor.defenses.hp > 0 ) { self.turn_order.push_back(turn); }
                self.turn_counter +=1;
                dbg!(&self.turn_counter);
                dbg!(&self.world);
            }
        }

        fn add_unit(&mut self, power : i64, defenses : Defenses) { 
            let new_id = self.world.actors.len() as u64;
            let actor = DamageableActor::new(new_id, power, defenses);
            self.world.actors.insert(actor.id, actor);
            self.add_listener(EventListener {
                order : 0,
                matcher : Self::attack_matcher(new_id),
                operator : Self::attack_operator(new_id)
            });
        }

        fn dispatch(&mut self, mut e : Event) {
            for listener in &self.listeners {
                if (listener.matcher)(&e) {
                    match (listener.operator)(&mut self.world,e){ 
                        Some(x) => e = x , 
                        None => break
                    }
                }
            }
        }

        fn add_listener(&mut self, listener : EventListener) {
            self.listeners.push(listener);
            //this could definitely be improved, probably by using a sorted vec or a tree
            self.listeners.sort_by(|a,b| a.order.cmp(&b.order));
        }

        ///match events that are attacks targeting this unit
        fn attack_matcher(self_id : u64) -> Box<dyn Fn(&Event) -> bool>{
            return Box::new( move |e : &Event| {
                match e.data {
                    EventData::Attack(_) => 
                        e.targets.iter().find(|u| **u==self_id).is_some(),
                    _ => false  
        }});}

        //event listener that damages this unit when an enemy attacks it
        fn attack_operator(self_id : u64) -> Box<dyn Fn(&mut World, Event) -> Option<Event>> {
            return Box::new( move |w : &mut World, e : Event| {
                if  let Some(this) = w.actors.get_mut(&self_id) && 
                    let EventData::Attack(dmg) = e.data 
                {
                    this.defenses.damage(&dmg);
                }
                return None;
            });
        }
    }
