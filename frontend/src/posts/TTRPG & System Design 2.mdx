    /// *the world can only be modified within operators*
    pub struct EventListener{
        order : i64,
        pub matcher : Box<dyn Fn(&Event) -> bool>,

    /// event system runs the game and routes events from actors to listeners. 
    pub struct EventSystem 
    {
        turn_order : VecDeque<u64>,
        world : World, 
        listeners : Vec<EventListener>,
        turn_counter : u64
        //plans : Vec
    }

    //an event system could be a thing thats constructed on combat start
    //from a subset of units in the world.
    impl EventSystem {
        fn start_combat(&mut self) {
            self.turn_counter = 0;
            //everything in the world will now engage in a battle to the death
            self.turn_order = self.world.actors.keys().map(|k|*k).collect::<VecDeque<u64>>();
            //lets just run till theres only one living unit
            while self.turn_order.len() > 1{
                let turn = self.turn_order.pop_front().unwrap();
                let mut actor = self.world.actors.get(&turn).expect("Actor with turn does not exist");
                if(0 > actor.defenses.hp) { continue ; }
                /// 'act' now returns a list of events
                let events = actor.act(&self.world);
                for event in events {
                    dbg!(&event);
                    ///the event system dispatches the returned events and executes listeners.
                    self.dispatch(event);
                }
                //need to relookup to shadow previous actor to make borrow checker happy.
                let mut actor = self.world.actors.get(&turn).expect("Actor with turn does not exist");
                if( actor.defenses.hp > 0 ) { self.turn_order.push_back(turn); }
                self.turn_counter +=1;
                dbg!(&self.turn_counter);
                dbg!(&self.world);
            }
        }

        fn add_unit(&mut self, power : i64, defenses : Defenses) { 
            let new_id = self.world.actors.len() as u64;
            let actor = DamageableActor::new(new_id, power, defenses);
            self.world.actors.insert(actor.id, actor);
            self.add_listener(EventListener {
                order : 0,
                matcher : Self::attack_matcher(new_id),
                operator : Self::attack_operator(new_id)
            });
        }

        fn dispatch(&mut self, mut e : Event) {
            for listener in &self.listeners {
                if (listener.matcher)(&e) {
                    match (listener.operator)(&mut self.world,e){ 
                        Some(x) => e = x , 
                        None => break
                    }
                }
            }
        }

        fn add_listener(&mut self, listener : EventListener) {
            self.listeners.push(listener);
            //this could definitely be improved, probably by using a sorted vec or a tree
            self.listeners.sort_by(|a,b| a.order.cmp(&b.order));
        }

        ///match events that are attacks targeting this unit
        fn attack_matcher(self_id : u64) -> Box<dyn Fn(&Event) -> bool>{
            return Box::new( move |e : &Event| {
                match e.data {
                    EventData::Attack(_) => 
                        e.targets.iter().find(|u| **u==self_id).is_some(),
                    _ => false  
        }});}

        //event listener that damages this unit when an enemy attacks it
        fn attack_operator(self_id : u64) -> Box<dyn Fn(&mut World, Event) -> Option<Event>> {
            return Box::new( move |w : &mut World, e : Event| {
                if  let Some(this) = w.actors.get_mut(&self_id) && 
                    let EventData::Attack(dmg) = e.data 
                {
                    this.defenses.damage(&dmg);
                }
                return None;
            });
        }
    }

I like this system though, it incorporates penetration in a unique way : 
Flat defense reduces the incoming damage by a flat amount while ratio does it by... a ratio.   
But only flat defense reduces the penetration of the incoming attack.  
So if an attack comes in with 5 damage and 5 penetration, it will completely bypass all ratio armor regardless of the number of layers.    
However if there is a layer with 6 flat defense on top, that will eat the 5 penetration and 1 of the damage, then 
the ratio armor will have full effect on the remaining damage. 

Penetration also doesn't care about damage type, it applies to whichever one is first.  
Think of it like a diamond tip to the attack.

This is dispatch. Now, I think in the chain of events that transpire from an event listener running,
it might need to dispatch more events! That's kinda the whole point. 

To dispatch events, it needs a &mut to the event system, which stores it. 

But iterating over the listeners requires at least a & to the event system, so we have an 
incompatible borrowing situation.

Conceptually this makes sense - we shouldn't be assuming we know all the listeners that exist 
before we've even run one of them. 

This doesn't change if I put the listeners on actors either. 

I think bevy's approach to this is to issue commands that are returned from the listener, then 
executed after iteration is done.

## What if we clone the whole world? 

Each listener needs a &mut to an EventSystem, but who's to say it has to be the same event system 
theyre in? 

And if we use Clone On Write smart pointers, maybe we can avoid actually cloning data the listener
doesnt change.

So what if we 
1. build a queue of listeners to run through 
2. for each one, do the following:
    1. clone the world (including the queue) and pass that to the listener 
    2. ~~pop and discard the first listener from the queue to prevent infinite recursion~~
    3. if the listener dispatches an event, we start from the beginning and recur automatically!
    4. we return a world for the parent event system to set it's world to. 

Wait how do we know when to stop running listeners? 
We need an ordering I guess but even then if we remove or reorder actors then whatever 
index we had into that ordering is meaningless. 

So actors and listeners cant be removed or reordered during event propagation, we have to defer that. 

Kinda seems like we need a little allocator or something for managing the list of actors.  

The actors need an 'active' flag, or maybe the listeners do, so we can deactivate them and not 
call them when a prior event listener would disable them without screwing up the listener queue.  

I'm not sure how to make the guarantee apparent to the type system that I'm not changing the order or amount
of the actors. Maybe using a range for iteration lets me out of that...

## COWS 

I don't know if cows are the answer but basically, the world is a `BTree<id,obj>`. 
I want to only clone the BTree if the listener modifies it.
However, I want to allow it to modify objects without having to clone the entire tree. 
Is this what cell's are for? 

Maybe `Cow<Btree<id, Cell<obj>>>` is what I'm looking for?

    ///match events that are attacks targeting this unit
    fn attack_matcher() -> Rc<dyn Fn(usize, &World, &Event) -> bool> {
        return Rc::new(move |id: usize, w: &World, e: &Event| match e.data {
            EventData::Attack(_) => e.targets.iter().find(|u| **u == id).is_some(),
            _ => false,
        });

    fn attack_operator() -> Rc<dyn Fn(usize, &mut World, Event) -> Option<Event>> {
        return Rc::new(move |id: usize, w: &mut World, e: Event| {
            if let Some(this) = w.actors.get_mut(&id)
            && let EventData::Attack(dmg) = &e.data
            {
                this.defenses.damage(&dmg.roll());
            }
            return None;
        });

        /// EventSystemCore has a default implementation of run_listeners that will execute them in the order they 
        /// were dispatched in. This can be altered if some other behavior is desired.
        /// TODO : should this return a result? 
        fn run_listeners(&mut self, event: Event) {
            self.core_mut().run_listeners(event);
        }
        /// EventSystemCore has a default implementation that should be used. 
        /// TODO : should this return a result?
        fn add_actor(&mut self, actor_data: A, behaviors: Vec<(&'static str, Option<BehaviorData>)>) -> Result<usize> {
            self.core_mut().add_actor(actor_data, behaviors)
        }

    }

    pub struct SequentialEventSystem<W, A>
        where A: ABounds<W>,
            W: WBounds
    {
        turn_order:   VecDeque<usize>,
        core : EventSystemCore<W,A>,
        turn_counter: u64, 
    }


    //an event system could be a thing thats constructed on combat start
    //from a subset of units in the world.
    //uses a 'plan -> execute' style where actor.act is only given a &world, and is expected to return a list of events.
    //an interactive system could be made where the same actor is allowed to act repeatedly until their turn ends.
    impl<W, A> EventSystem<W, A> for SequentialEventSystem<W,A>
        where A: ABounds<W>,
            W: WBounds
    {
        fn new() -> Self {
            let turn_order = VecDeque::new();
            let core = EventSystemCore::new();
            let turn_counter = 0;
            return SequentialEventSystem { turn_order, core, turn_counter};
        }

        fn core(&self) -> &EventSystemCore<W,A> {
            return &self.core;
        }

        fn core_mut(&mut self)-> &mut EventSystemCore<W,A> {
            return &mut self.core;
        }

        fn start(&mut self) {
            self.turn_order = self.core.world.actors.keys().map(|k| *k).collect::<VecDeque<usize>>();
        }

        fn next(&mut self) {
            //run the next turn
            let turn = self.turn_order.pop_front().unwrap();
            let mut actor = self.core
                                .world
                                .actors

    use std::{any::{Any, TypeId}, collections::BTreeMap, rc::Rc};

    use rand::random_range;

    use crate::event_sys::*;

    const TAKE_DAMAGE : &'static str = "TakeDamage";

    #[derive(Debug)]
    struct ActorData {
        hp : i64
    }

    impl ActorData {
        fn goblin() -> Self {
            return Self{hp:4}
        }
        fn knight() -> Self {
            return Self{hp:8}
        }
    }

    #[derive(Default, Debug)]
    struct WorldData {}

    struct EventData { damage : i64 }

    impl Act<WorldData,Self> for ActorData {
        fn init(&self, self_id: usize, world: &World<WorldData, Self>, behaviors: &BehaviorMap<WorldData, Self>) -> Option<Event> {
            return None
        }

        fn act(&self, self_id: usize, world: &World<WorldData, Self>, behaviors: &BehaviorMap<WorldData, Self>) -> Option<Event> {        
            //find an actor in the world that isnt self
            let others = world.iter().filter( |a| a.0 != &self_id ).map(|a| *a.0).collect::<Vec<usize>>();
            let random_choice = others.get(random_range(0..others.len()));
            if let Some(id) = random_choice {
                //send a 'damage' event to it if we find one.
                return Some(Event::new(self_id, vec![*id], Box::new( EventData { damage : 1 } )));
            }
            else {
                return None;
            }
        }
    }

    impl ActorBehaviors<WorldData,Self> for ActorData {
        fn all_behaviors() -> BehaviorMap<WorldData,Self> {
            let mut map = BehaviorMap::new();
            let mut matchers = BTreeMap::new();
            matchers.insert( TypeId::of::<EventData>(), take_damage_matcher as Matcher<WorldData,Self>);
            map.insert(TAKE_DAMAGE, Behavior {
                initializer : take_damage_intializer,
                matchers,
                order: 0,
                statuses : vec![],
                operator : take_damage_operator
            });
            return map;
        }
    }

    struct TakeDamageData();

    fn take_damage_matcher(id : usize, w : &World<WorldData,ActorData>, event : &Event, behavior_data : Option<&BehaviorData>) -> bool {
        //we can assume event is the correct type due to how listeners work
        return event.targets.iter().find(|&target| target==&id).is_some() 
    }

    //if would be nice if operator could see what behaviors exist...
    //also unwrap is too tempting when a result cant be returned, best to make this return a result.
    fn take_damage_operator(id : usize, w : &mut World<WorldData,ActorData>, d : &mut Dispatcher, event : Event) -> Option<Event>{
        let evt_data : &EventData = event.data.downcast_ref::<EventData>().expect("Operator was passed an event with the wrong type_id");
        let actor = w.get_actor_mut(id).expect("Actor not found");
        actor.inner.hp -= evt_data.damage;
        return Some(event);
    }

    /// this behavior doesn't use special data so we just make up a zero sized type.
    fn take_damage_intializer(o : Option<BehaviorData>) -> Box<dyn Any> {
        return Box::new(TakeDamageData);
    }
